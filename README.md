# how to create a virtual environment in python
'''conda create -n env name python=3.10(version that you want) -y

For eg: 
conda create -n Final_project python=3.10 -y'''


# how to activate the environment 
'''conda activate Final_project'''

# command to install the required package
step 1: create a setup.py file
step 2: In your bash, pip install -r file name where you have kept the package
eg: pip install -r requirements.txt
